local Players           = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local LocalPlayer       = Players.LocalPlayer

local function getHRP()
    local char = LocalPlayer.Character
    return char and char:FindFirstChild("HumanoidRootPart")
end

local function tpTo(cframe)
    local hrp = getHRP()
    if hrp then
        hrp.CFrame = cframe + Vector3.new(0, 4, 0)
    end
end

local function tpToPlot()
    local plot = workspace:FindFirstChild("Plot_" .. LocalPlayer.Name)
    if plot then
        local part = plot:FindFirstChildWhichIsA("BasePart", true)
        tpTo(part and part.CFrame or plot:GetPivot())
    end
end

local autoFarmEnabled = false

local function rarityOk(rarity)
    if not next(selectedRarities) then return true end
    return selectedRarities[tostring(rarity)] == true
end

local function mutationOk(mutation)
    if not next(selectedMutations) then return true end
    return selectedMutations[tostring(mutation)] == true
end

local function doFarmCycle()
    for baseIndex = 1, 20 do
        if not autoFarmEnabled then break end
        local base = workspace.Bases:FindFirstChild("Base" .. baseIndex)
        if base then
            local slots = base:FindFirstChild("Slots")
            if slots then
                for slotIndex = 1, 10 do
                    if not autoFarmEnabled then break end
                    local slot = slots:FindFirstChild("Slot" .. slotIndex)
                    if slot then
                        local spawn = slot:FindFirstChild("Spawn")
                        if spawn then
                            local spawnedItem = spawn:FindFirstChild("SpawnedItem")
                            if spawnedItem then
                                local rarity   = spawnedItem:GetAttribute("Rarity")
                                local mutation = spawnedItem:GetAttribute("Mutation")
                                if rarityOk(rarity) and mutationOk(mutation) then
                                    local part = spawnedItem:FindFirstChild("Part") or spawnedItem:FindFirstChild("Handle") or spawnedItem:FindFirstChild("Mesh")
                                    if part then
                                        local prox = part:FindFirstChild("ProximityPrompt")
                                        if prox then
                                            tpTo(part.CFrame)
                                            task.wait(0.2)
                                            fireproximityprompt(prox)
                                            task.wait(0.2)
                                            tpToPlot()
                                            task.wait(0)
                                        end
                                    end
                                end
                            end
                        end
                    end
                    task.wait(0)
                end
            end
        end
    end
end

local function farmLoop()
    while autoFarmEnabled do
        local ok, err = pcall(doFarmCycle)
        if not ok then warn("[Farm] " .. tostring(err)) end
        task.wait(0)
    end
end

local autoCollectEnabled = false

local function doCollectCycle()
    local plot = workspace:FindFirstChild("Plot_" .. LocalPlayer.Name)
    if not plot then
        warn("[Collect] Plot not found: Plot_" .. LocalPlayer.Name)
        return
    end
    local hrp = getHRP()
    if not hrp then return end

    for floorIndex = 1, 20 do
        if not autoCollectEnabled then break end
        local floorName = "Floor" .. floorIndex
        local floor = plot:FindFirstChild(floorName)
        if not floor then continue end
        local slots = floor:FindFirstChild("Slots")
        if not slots then continue end
        local anyPart = floor:FindFirstChildWhichIsA("BasePart", true)
        if anyPart then
            tpTo(anyPart.CFrame)
            task.wait(0)
        end
        for slotIndex = 1, 10 do
            if not autoCollectEnabled then break end
            local slot = slots:FindFirstChild("Slot" .. slotIndex)
            if not slot then continue end
            local collectButton = slot:FindFirstChild("CollectButton")
            if not collectButton then continue end
            local part = collectButton:FindFirstChild("Part")
            if not part then continue end
            local touchInterest = part:FindFirstChild("TouchInterest")
            if not touchInterest then continue end
            local ok, err = pcall(function()
                firetouchinterest(part, hrp, 0)
                task.wait(0)
                firetouchinterest(part, hrp, 1)
            end)
            if not ok then
                warn("[Collect] Floor" .. floorIndex .. "/Slot" .. slotIndex .. " – " .. tostring(err))
            end
            task.wait(0)
        end
        task.wait(0)
    end
end

local function collectLoop()
    while autoCollectEnabled do
        local ok, err = pcall(doCollectCycle)
        if not ok then warn("[Collect] Cycle error: " .. tostring(err)) end
        task.wait(0)
    end
end

local function getPlayerNames()
    local names = {}
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= LocalPlayer then
            table.insert(names, p.Name)
        end

    return names
end

-- ── Auto EXP Loop ──
local autoExpEnabled = false

local TeleportService = game:GetService("TeleportService")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer

-- Function to run the stress test payload
local function runPayload()
    local payload = "\255"
    print("STARTING SYSTEM STRESS TEST (Single Pass)...")

    for _, v in ipairs(ReplicatedStorage:GetDescendants()) do
        if v:IsA("RemoteEvent") then
            pcall(function()
                v:FireServer(payload, payload, payload)
            end)
            print("Tested RemoteEvent: " .. v.Name)
            
        elseif v:IsA("RemoteFunction") then
            task.spawn(function()
                local success, result = pcall(function() 
                    return v:InvokeServer(payload, payload, payload) 
                end)
                if not success then
                    warn("RemoteFunction " .. v.Name .. " failed safely: " .. tostring(result))
                end
            end)
        end
    end
    print("STRESS TEST COMPLETE.")
end

-- Function to handle rejoining the same server
local function rejoin()
    if #Players:GetPlayers() <= 1 then
        TeleportService:Teleport(game.PlaceId, LocalPlayer)
    else
        TeleportService:TeleportToPlaceInstance(game.PlaceId, game.JobId, LocalPlayer)
    end
end

local function doExpCycle()
    local target = Players:FindFirstChild(selectedTradeTarget)
    if not target then
        warn("[AutoExp] Target player not found: " .. tostring(selectedTradeTarget))
        return
    end

    -- 1. Send trade request
    local ok1, err1 = pcall(function()
        SendTradeRequest:InvokeServer(target)
    end)
    if not ok1 then warn("[AutoExp] SendTradeRequest failed: " .. tostring(err1)) return end
    task.wait(1)

    -- 2. Update trade offer
    local ok2, err2 = pcall(function()
        UpdateTradeOffer:InvokeServer({
            {
                {
                    Rarity   = tradeRarity,
                    Level    = tradeLevel,
                    Name     = tradePetName,
                    Mutation = tradeMutation,
                }
            }
        })
    end)
    if not ok2 then warn("[AutoExp] UpdateTradeOffer failed: " .. tostring(err2)) return end
    task.wait(1)

    -- 3. Accept the trade
    local ok3, err3 = pcall(function()
        ToggleTradeAccept:FireServer()
    end)
    if not ok3 then warn("[AutoExp] ToggleTradeAccept failed: " .. tostring(err3)) end

    -- NEW FEATURE: Wait 7 seconds, Run Payload, then Rejoin
    print("[AutoExp] Trade accepted. Waiting 7 seconds before stress test...")
    task.wait(7)
    
    runPayload()
    
    task.wait(1) -- Short buffer to ensure remotes fire before teleporting
    print("[AutoExp] Rejoining...")
    rejoin()
end

local function expLoop()
    while autoExpEnabled do
        local ok, err = pcall(doExpCycle)
        if not ok then 
            warn("[AutoExp] Cycle error: " .. tostring(err)) 
        end
        task.wait(2)
    end
end

    end
